\label{app:tully_params}
\section{Model 1 -Single Avoided Crossing}
\begin{flalign*}
  H_{11}(\mathbf{R}) \ &= \ A \ \tanh(B\mathbf{R}) &\\
  H_{12}(\mathbf{R}) \ &= \ C e^{-D\mathbf{R}^2} \\
  H_{21}(\mathbf{R}) \ &= \ H_{12}(\mathbf{R}) \\
  H_{22}(\mathbf{R}) \ &= \ -H_{11}(\mathbf{R})
\end{flalign*}
Where A = 0.03, B = 0.4, C = 0.005 and D = 0.3

\section{Model 2 -Dual Avoided Crossing}
\begin{flalign*}
  H_{11}(\mathbf{R}) \ &= \ 0 &\\
  H_{12}(\mathbf{R}) \ &= \ C e^{-D \mathbf{R}^2} \\
  H_{21}(\mathbf{R}) \ &= \ H_{12}(\mathbf{R}) \\
  H_{22}(\mathbf{R}) \ &= \ -A e^{-B\mathbf{R}^2} + E
\end{flalign*}
Where A = 0.1, B = 0.28, C = 0.015, D = 0.06 and E = 0.05

\section{Model 3 -Extended Coupling}
\begin{flalign*}
  H_{11}(\mathbf{R}) \ &= \ A  &\\
  H_{12}(\mathbf{R}) \ &= \ \left \lbrace
  \begin{array}{l}
    B e^{C\mathbf{R}}, \qquad \qquad \qquad R \leq 0 \\
      B(2 - e^{-C\mathbf{R}}),  \qquad \quad R > 0\\
  \end{array} \right . \\
  H_{21}(\mathbf{R}) \ &= \ H_{12}(\mathbf{R}) \\
  H_{22}(\mathbf{R}) \ &= \ -H_{11}(\mathbf{R})
\end{flalign*}
Where A = 6$\times$ 10$^{-4}$, B = 0.1 and C = 0.9

\section{Model 4 -Dual Arch}
\begin{flalign*}
  H_{11}(\mathbf{R}) \ &= \ A  &\\
  H_{12}(\mathbf{R}) \ &= \ \left \lbrace
  \begin{array}{l}
    B \left( -e^{C(\mathbf{R} - D)} + e^{C(\mathbf{R} + D)} \right), \qquad \qquad \ \ \qquad \quad \ \ R \leq -D \\
    B \left( e^{-C(\mathbf{R} - D)} - e^{-C(\mathbf{R} + D)} \right), \qquad \qquad \ \ \qquad \quad R \geq D \\
    B \left( 2 - e^{C(\mathbf{R} - D)} - e^{-C(\mathbf{R} + D)} \right), \qquad \qquad -D < R < D \\
  \end{array} \right . \\
  H_{21}(\mathbf{R}) \ &= \ H_{12}(\mathbf{R}) \\
  H_{22}(\mathbf{R}) \ &= \ -H_{11}(\mathbf{R})
\end{flalign*}
Where A = 6$\times$ 10$^{-4}$, B = 0.1 and C = 0.9
